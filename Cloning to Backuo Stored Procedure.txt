CREATE OR REPLACE PROCEDURE TRUSTBANKDATA.HARMONIZED.CLONE_TABLES_TO_BACKUP()
RETURNS VARCHAR(16777216)
LANGUAGE SQL
EXECUTE AS OWNER
AS '
BEGIN
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.ATM_LOOKUP CLONE HARMONIZED.ATM_LOOKUP'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.BRANCH CLONE HARMONIZED.BRANCH'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.HOUR_LOOKUP CLONE HARMONIZED.HOUR_LOOKUP'';
    
   
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.BRANCH_LOOKUP CLONE HARMONIZED.BRANCH_LOOKUP'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.CUSTOMER_LOOKUP CLONE HARMONIZED.CUSTOMER_LOOKUP'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.SYSTEM_HEALTH_LOGS CLONE HARMONIZED.SYSTEM_HEALTH_LOGS'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.TRANSACTIONS CLONE HARMONIZED.TRANSACTIONS'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.LOAN CLONE HARMONIZED.LOAN'';
    
  
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.BRANCH_PERFORMANCE CLONE HARMONIZED.BRANCH_PERFORMANCE'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.INVESTMENTS CLONE HARMONIZED.INVESTMENTS'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.TRANSACTIONS_TYPE_LOOKUP CLONE HARMONIZED.TRANSACTIONS_TYPE_LOOKUP'';
    

    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.CALENDER_LOOKUP CLONE HARMONIZED.CALENDER_LOOKUP'';
    
    EXECUTE IMMEDIATE 
    ''CREATE OR REPLACE TABLE BACKUP.CUSTOMER_DEMO CLONE HARMONIZED.CUSTOMER_DEMO'';

    RETURN ''Cloning completed successfully at '' || CURRENT_TIMESTAMP();
END;
';
